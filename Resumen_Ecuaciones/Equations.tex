        \hline

        \hline 
        $\sin (2 \alpha) = 2 \cdot \sin (\alpha) \cdot \cos (\alpha)$ \\

        $\sin^2 (\alpha) = \frac{1}{2} \cdot \left ( 1 - \cos (2 \alpha) \right )$ \\
        \hline
